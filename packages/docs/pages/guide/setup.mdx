# Project Setup

This guide walks you through creating a standalone e2e-tests package that imports
`power-platform-playwright-toolkit` and `playwright-ms-auth` to test Power Platform applications.

<Callout type="info">
  This guide covers setting up a **standalone npm project** (not a monorepo). If you're inside a
  Rush/Nx monorepo, the same principles apply — just adapt the dependency declarations.
</Callout>

## Prerequisites

| Requirement | Version | Notes |
|------------|---------|-------|
| Node.js | 20+ | LTS recommended |
| npm / pnpm / yarn | latest | Any package manager |
| Microsoft Edge | latest | Required for Power Platform |
| Power Platform access | — | Test account with no MFA |

<Steps>

### Initialise the project

```bash
mkdir my-power-platform-e2e && cd my-power-platform-e2e
npm init -y
```

### Create `playwright.config.ts`

```typescript
import * as path from 'path';
import { defineConfig } from '@playwright/test';
import dotenv from 'dotenv';
import { getStorageStatePath, TimeOut, ConfigHelper } from 'power-platform-playwright-toolkit';

dotenv.config({ path: path.resolve(__dirname, '.env') });

export default defineConfig({
  globalSetup: require.resolve('./globals/global-setup'),
  globalTeardown: require.resolve('./globals/global-teardown'),

  fullyParallel: !process.env.CI,
  forbidOnly: !!process.env.CI,
  retries: process.env.CI ? 1 : 0,
  workers: process.env.CI ? 1 : 2,
  timeout: TimeOut.TestTimeout,

  testDir: './tests',
  testMatch: ['**/*.test.ts'],

  outputDir: path.join('./test-results', 'artifacts'),

  expect: {
    timeout: TimeOut.DefaultWaitTime,
  },

  reporter: process.env.CI
    ? [
        ['dot'],
        ['blob', { outputDir: './test-results/blob-report' }],
        ['junit', { outputFile: './test-results/artifacts/testResults/results.xml' }],
      ]
    : [
        ['list'],
        ['html', { outputFolder: './test-results/html-report', open: 'never' }],
      ],

  use: {
    baseURL: ConfigHelper.getBaseUrl(),
    channel: 'msedge',
    headless: process.env.CI ? true : process.env.HEADLESS === 'true',
    viewport: { width: 1920, height: 1080 },
    screenshot: 'only-on-failure',
    video: 'retain-on-failure',
    trace: 'retain-on-failure',
    actionTimeout: TimeOut.OneMinuteTimeOut,
    navigationTimeout: TimeOut.OneMinuteTimeOut,
    ignoreHTTPSErrors: true,
    acceptDownloads: true,
    locale: 'en-US',
    permissions: ['clipboard-read', 'clipboard-write'],
    launchOptions: {
      args: ['--start-maximized', '--window-size=1920,1080'],
    },
  },

  projects: [
    {
      name: 'model-driven-app',
      testMatch: '**/mda/**/*.test.ts',
      use: {
        storageState: process.env.MS_AUTH_EMAIL
          ? path.join(
              path.dirname(getStorageStatePath(process.env.MS_AUTH_EMAIL)),
              `state-mda-${process.env.MS_AUTH_EMAIL}.json`
            )
          : undefined,
      },
    },
    {
      name: 'canvas-app',
      testMatch: '**/canvas/**/*.test.ts',
      use: {
        storageState: process.env.MS_AUTH_EMAIL
          ? getStorageStatePath(process.env.MS_AUTH_EMAIL)
          : undefined,
      },
    },
  ],
});
```

### Create the authentication script

`scripts/authenticate.ts`:

```typescript
import { chromium } from '@playwright/test';
import {
  authenticateToMicrosoft,
  loadAuthConfig,
  getStorageStatePath,
} from 'power-platform-playwright-toolkit';
import * as path from 'path';
import * as dotenv from 'dotenv';

dotenv.config({ path: path.resolve(__dirname, '../.env') });

const isMda = process.argv.includes('--mda');
const isHeadful = process.argv.includes('--headful');

async function authenticate() {
  const config = loadAuthConfig();
  const browser = await chromium.launch({ headless: !isHeadful, channel: 'msedge' });
  const context = await browser.newContext();
  const page = await context.newPage();

  console.log(`Authenticating: ${config.email} (${isMda ? 'MDA' : 'default'})`);
  await authenticateToMicrosoft(page, config);

  const base = getStorageStatePath(config.email);
  const statePath = isMda
    ? path.join(path.dirname(base), `state-mda-${config.email}.json`)
    : base;

  await context.storageState({ path: statePath });
  await browser.close();
  console.log(`Saved: ${statePath}`);
}

authenticate().catch((e) => { console.error(e); process.exit(1); });
```

### Create a Page Object Model

`pages/northwind/NorthwindMda.page.ts`:

```typescript
import { Page, BrowserContext } from '@playwright/test';
import {
  AppProvider,
  AppType,
  AppLaunchMode,
  ModelDrivenAppPage,
} from 'power-platform-playwright-toolkit';

const APP_URL = process.env.MODEL_DRIVEN_APP_URL!;
const ENTITY = 'nwind_orders';

export class NorthwindMdaPage {
  readonly appProvider: AppProvider;
  mda!: ModelDrivenAppPage;

  constructor(page: Page, context: BrowserContext) {
    this.appProvider = new AppProvider(page, context);
  }

  async launch() {
    await this.appProvider.launch({
      app: 'Northwind Orders',
      type: AppType.ModelDriven,
      mode: AppLaunchMode.Play,
      skipMakerPortal: true,
      directUrl: APP_URL,
    });
    this.mda = this.appProvider.getModelDrivenAppPage();
  }

  async navigateToOrders() {
    await this.mda.navigateToGridView(ENTITY);
    await this.mda.grid.waitForGridLoad();
  }

  async openOrder(rowIndex: number) {
    await this.mda.grid.openRecord({ rowNumber: rowIndex });
  }
}
```
